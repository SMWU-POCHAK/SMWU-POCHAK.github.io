        final Long postId,
        final Long loginMemberId
) {

            query.selectFrom(post)
                    .join(post.owner).fetchJoin()
                    .join(tag).on(tag.post.eq(post))

                            checkOwnerOrTaggedMemberBlockLoginMember(loginMemberId)

                    .having(block.id.count().eq(0L))
